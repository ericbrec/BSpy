  8class _BitCheckbutton(tk.Checkbutton):
  9    """A tkinter `CheckButton` that gets/sets its variable based on a given `bitmask`."""
 10    def __init__(self, parent, bitMask, **kw):
 11        self.bitMask = bitMask
 12        self.variable = kw.get("variable")
 13        self.command = kw.get("command")
 14        self.var = tk.IntVar()
 15        self.var.set(1 if self.variable.get() & self.bitMask else 0)
 16        kw["variable"] = self.var
 17        kw["onvalue"] = 1
 18        kw["offvalue"] = 0
 19        kw["command"] = self.Command
 20        tk.Checkbutton.__init__(self, parent, **kw)
 21    
 22    def Command(self):
 23        """
 24        Handles when the checkbutton is pushed, updating the variable based on the 
 25        bitmask and then calling the checkbutton command.
 26        """
 27        if self.var.get() == 1:
 28            self.variable.set(self.variable.get() | self.bitMask)
 29        else:
 30            self.variable.set(self.variable.get() & ~self.bitMask)
 31        self.command(self.variable.get())
 32
 33    def Update(self):
 34        """Updates checkbutton state."""
 35        self.var.set(1 if self.variable.get() & self.bitMask else 0)

111    def _check_for_work(self):
112        """Check queue for calls to make."""
113        while not self.workQueue.empty():
114            try:
115                work = self.workQueue.get_nowait()
116            except queue.Empty:

289    def _app_thread(self):
290        app = bspyApp(workQueue=self.workQueue)
