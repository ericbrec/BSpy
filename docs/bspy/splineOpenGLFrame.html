  22        void ComputeBSpline(in int offset, in int order, in int n, in int knot, in float u, 
  23            out float uBSpline[{maxOrder}], out float duBSpline[{maxOrder}])
  24        {{
  25            int degree = 1;
  26
  27            for (int i = 0; i < {maxOrder}; i++)
  28            {{
  29                uBSpline[i] = 0.0;
  30                duBSpline[i] = 0.0;
  31            }}
  32            uBSpline[order-1] = 1.0;
  33
  34            while (degree < order - 1)
  35            {{
  36                int b = order - degree;
  37                for (int i = knot - degree; i < knot; i++)
  38                {{
  39                    float knotValue = texelFetch(uSplineData, offset + i).x; // knots[i]
  40                    float alpha = (u - knotValue) / (texelFetch(uSplineData, offset + i + degree).x - knotValue); // (u - knots[i]) / (knots[i+degree] - knots[i]);
  41                    uBSpline[b-1] += (1.0 - alpha) * uBSpline[b];
  42                    uBSpline[b] *= alpha;
  43                    b++;
  44                }}
  45                degree++;
  46            }}
  47            if (degree < order)
  48            {{
  49                int b = order - degree;
  50                for (int i = knot - degree; i < knot; i++)
  51                {{
  52                    float knotValue = texelFetch(uSplineData, offset + i).x; // knots[i]
  53                    float gap = texelFetch(uSplineData, offset + i + degree).x - knotValue; // knots[i+degree] - knots[i]
  54                    float alpha = degree / gap;
  55                    duBSpline[b-1] += -alpha * uBSpline[b];
  56                    duBSpline[b] = alpha * uBSpline[b];
  57
  58                    alpha = (u - knotValue) / gap; // (u - knots[i]) / gap;
  59                    uBSpline[b-1] += (1.0 - alpha) * uBSpline[b];
  60                    uBSpline[b] *= alpha;
  61                    b++;
  62                }}
  63            }}
  64        }}

  67    computeSampleRateCode = """
  68        float ComputeSampleRate(in vec4 point, in vec3 dPoint, in vec3 d2Point, in float minRate)
  69        {
  70            float rate = 0.0;
  71            float scale = uScreenScale.z > 0.0 ? -point.z : 1.0;
  72
  73            // Only consider points that lie within the clip bounds or whose derivative spans the clip bounds.
  74            if (point.z < uClipBounds[3] && point.z > uClipBounds[2] && 
  75                ((point.y < scale * uClipBounds[1] && point.y > -scale * uClipBounds[1]) ||
  76                (point.y >= scale * uClipBounds[1] && point.y + dPoint.y <= -scale * uClipBounds[1]) ||
  77                (point.y <= -scale * uClipBounds[1] && point.y + dPoint.y >= scale * uClipBounds[1])) &&
  78                ((point.x < scale * uClipBounds[0] && point.x > -scale * uClipBounds[0]) ||
  79                (point.x >= scale * uClipBounds[0] && point.x + dPoint.x <= -scale * uClipBounds[0]) ||
  80                (point.x <= -scale * uClipBounds[0] && point.x + dPoint.x >= scale * uClipBounds[0])))
  81            {
  82                float zScale = -1.0 / point.z;
  83                float zScale2 = zScale * zScale;
  84                float zScale3 = zScale2 * zScale;
  85                vec2 projection = uScreenScale.z > 0.0 ? 
  86                    vec2(uScreenScale.x * (d2Point.x * zScale - 2.0 * dPoint.x * dPoint.z * zScale2 +
  87                        point.x * (2.0 * dPoint.z * dPoint.z * zScale3 - d2Point.z * zScale2)),
  88                        uScreenScale.y * (d2Point.y * zScale - 2.0 * dPoint.y * dPoint.z * zScale2 +
  89                        point.y * (2.0 * dPoint.z * dPoint.z * zScale3 - d2Point.z * zScale2)))
  90                    : vec2(uScreenScale.x * d2Point.x, uScreenScale.y * d2Point.y);
  91                rate = max(sqrt(length(projection)), minRate);
  92            }
  93            return rate;
  94        }
  95    """

 126    computeCurveSamplesCode = """
 127        void ComputeCurveSamples(in int maxSamples, out float uSamples)
 128        {
 129            float sampleRate = 0.0;
 130            if (outData.uInterval > 0.0)
 131            {
 132                float minRate = 1.0 / outData.uInterval;
 133                if (outData.uOrder < 3)
 134                {
 135                    // It's a line or point, so just do the minimum sample.
 136                    sampleRate = minRate;
 137                }
 138                else
 139                {
 140                    int i = outData.uKnot - outData.uOrder;
 141                    int coefficientOffset = header + outData.uOrder + outData.uN + 4 * i;
 142                    vec4 coefficient0 = vec4(
 143                        texelFetch(uSplineData, coefficientOffset+0).x, 
 144                        texelFetch(uSplineData, coefficientOffset+1).x,
 145                        texelFetch(uSplineData, coefficientOffset+2).x,
 146                        texelFetch(uSplineData, coefficientOffset+3).x);
 147                    coefficientOffset += 4;
 148                    vec4 coefficient1 = vec4(
 149                        texelFetch(uSplineData, coefficientOffset+0).x, 
 150                        texelFetch(uSplineData, coefficientOffset+1).x,
 151                        texelFetch(uSplineData, coefficientOffset+2).x,
 152                        texelFetch(uSplineData, coefficientOffset+3).x);
 153                    float gap = texelFetch(uSplineData, header + i+outData.uOrder).x - texelFetch(uSplineData, header + i+1).x; // uKnots[i+uOrder] - uKnots[i+1]
 154                    vec3 dPoint0 = ((outData.uOrder - 1) / gap) * (coefficient1.xyz - coefficient0.xyz);
 155                    while (i < outData.uKnot-2)
 156                    {
 157                        coefficientOffset += 4;
 158                        vec4 coefficient2 = vec4(
 159                            texelFetch(uSplineData, coefficientOffset+0).x, 
 160                            texelFetch(uSplineData, coefficientOffset+1).x,
 161                            texelFetch(uSplineData, coefficientOffset+2).x,
 162                            texelFetch(uSplineData, coefficientOffset+3).x);
 163                        gap = texelFetch(uSplineData, header + i+1+outData.uOrder).x - texelFetch(uSplineData, header + i+2).x; // uKnots[i+1+uOrder] - uKnots[i+2]
 164                        vec3 dPoint1 = ((outData.uOrder - 1) / gap) * (coefficient2.xyz - coefficient1.xyz);
 165                        gap = texelFetch(uSplineData, header + i+outData.uOrder).x - texelFetch(uSplineData, header + i+2).x; // uKnots[i+uOrder] - uKnots[i+2]
 166                        vec3 d2Point = ((outData.uOrder - 2) / gap) * (dPoint1 - dPoint0);
 167
 168                        sampleRate = max(sampleRate, ComputeSampleRate(coefficient0, dPoint0, d2Point, minRate));
 169                        sampleRate = max(sampleRate, ComputeSampleRate(coefficient1, dPoint0, d2Point, minRate));
 170                        sampleRate = max(sampleRate, ComputeSampleRate(coefficient1, dPoint1, d2Point, minRate));
 171                        sampleRate = max(sampleRate, ComputeSampleRate(coefficient2, dPoint1, d2Point, minRate));
 172
 173                        coefficient0 = coefficient1;
 174                        coefficient1 = coefficient2;
 175                        dPoint0 = dPoint1;
 176                        i++;
 177                    }
 178                }
 179            }
 180            uSamples = min(floor(0.5 + outData.uInterval * sampleRate), maxSamples);
 181        }

 410        void ComputeSurfaceSamples(in int maxSamples, out float uSamples[3], out float vSamples[3])
 411        {{
 412            // Computes sample counts for u and v for the left side ([0]), middle ([1]), and right side ([2]).
 413            // The left side sample count matches the right side sample count for the previous knot.
 414            // The middle sample count is the number of samples between knots (same as ComputeCurveSamples).
 415            float sampleRate[3] = float[3](0.0, 0.0, 0.0);
 416            if (outData.uInterval > 0.0)
 417            {{
 418                float minRate = 1.0 / outData.uInterval;
 419                if (outData.uOrder < 3)
 420                {{
 421                    // It's a plane or point, so just do the minimum sample.
 422                    sampleRate = float[3](minRate, minRate, minRate);
 423                }}
 424                else
 425                {{
 426                    float sampleRateLeft[{maxOrder}];
 427                    float sampleRateRight[{maxOrder}];
 428
 429                    for (int k = 0; k < outData.uOrder; k++)
 430                    {{
 431                        sampleRateLeft[k] = 0.0;
 432                        sampleRateRight[k] = 0.0;
 433                    }}
 434                    for (int j = outData.vKnot-outData.vOrder; j < outData.vKnot; j++)
 435                    {{
 436                        int i = max(outData.uKnot - 1 - outData.uOrder, 0);
 437                        int iLimit = min(outData.uKnot - 1, outData.uN - 2);
 438                        int coefficientOffset = header + outData.uOrder+outData.uN + outData.vOrder+outData.vN + 4*outData.uN*j + 4*i;
 439                        vec4 coefficient0 = vec4(
 440                            texelFetch(uSplineData, coefficientOffset+0).x, 
 441                            texelFetch(uSplineData, coefficientOffset+1).x,
 442                            texelFetch(uSplineData, coefficientOffset+2).x,
 443                            texelFetch(uSplineData, coefficientOffset+3).x);
 444                        coefficientOffset += 4;
 445                        vec4 coefficient1 = vec4(
 446                            texelFetch(uSplineData, coefficientOffset+0).x, 
 447                            texelFetch(uSplineData, coefficientOffset+1).x,
 448                            texelFetch(uSplineData, coefficientOffset+2).x,
 449                            texelFetch(uSplineData, coefficientOffset+3).x);
 450                        float gap = texelFetch(uSplineData, header + i+outData.uOrder).x - texelFetch(uSplineData, header + i+1).x; // uKnots[i+uOrder] - uKnots[i+1]
 451                        vec3 dPoint0 = ((outData.uOrder - 1) / gap) * (coefficient1.xyz - coefficient0.xyz);
 452                        while (i < iLimit)
 453                        {{
 454                            coefficientOffset += 4;
 455                            vec4 coefficient2 = vec4(
 456                                texelFetch(uSplineData, coefficientOffset+0).x, 
 457                                texelFetch(uSplineData, coefficientOffset+1).x,
 458                                texelFetch(uSplineData, coefficientOffset+2).x,
 459                                texelFetch(uSplineData, coefficientOffset+3).x);
 460                            gap = texelFetch(uSplineData, header + i+1+outData.uOrder).x - texelFetch(uSplineData, header + i+2).x; // uKnots[i+1+uOrder] - uKnots[i+2]

 463                            vec3 d2Point = ((outData.uOrder - 2) / gap) * (dPoint1 - dPoint0);
 464

 466                            sampleRateLeft[k] = max(sampleRateLeft[k], ComputeSampleRate(coefficient0, dPoint0, d2Point, minRate));
 467                            sampleRateLeft[k] = max(sampleRateLeft[k], ComputeSampleRate(coefficient1, dPoint0, d2Point, minRate));
 468                            sampleRateRight[k] = max(sampleRateRight[k], ComputeSampleRate(coefficient1, dPoint1, d2Point, minRate));
 469                            sampleRateRight[k] = max(sampleRateRight[k], ComputeSampleRate(coefficient2, dPoint1, d2Point, minRate));

 473                            dPoint0 = dPoint1;

 475                        }}
 476                    }}

 482                        sampleRate[1] = max(sampleRate[1], sampleRateRight[k]);

 485                    }}
 486                }}

 489            uSamples[1] = min(floor(0.5 + outData.uInterval * sampleRate[1]), maxSamples);
 490            uSamples[2] = min(floor(0.5 + outData.uInterval * sampleRate[2]), maxSamples);
 491

 493            if (outData.vInterval > 0.0)

 552                        }}
 553                    }}

 567            vSamples[2] = min(floor(0.5 + outData.vInterval * sampleRate[2]), maxSamples);

 411        void ComputeSurfaceSamples(in int maxSamples, out float uSamples[3], out float vSamples[3])

 413            // Computes sample counts for u and v for the left side ([0]), middle ([1]), and right side ([2]).
 414            // The left side sample count matches the right side sample count for the previous knot.
 415            // The middle sample count is the number of samples between knots (same as ComputeCurveSamples).
 416            float sampleRate[3] = float[3](0.0, 0.0, 0.0);
 417            if (outData.uInterval > 0.0)
 418            {{
 419                float minRate = 1.0 / outData.uInterval;
 420                if (outData.uOrder < 3)
 421                {{
 422                    // It's a plane or point, so just do the minimum sample.
 423                    sampleRate = float[3](minRate, minRate, minRate);
 424                }}
 425                else
 426                {{
 427                    float sampleRateLeft[{maxOrder}];
 428                    float sampleRateRight[{maxOrder}];
 429
 430                    for (int k = 0; k < outData.uOrder; k++)
 431                    {{
 432                        sampleRateLeft[k] = 0.0;
 433                        sampleRateRight[k] = 0.0;
 434                    }}
